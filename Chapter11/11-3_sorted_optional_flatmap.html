sorted()：ストリームをソートする

sorted の使い方
import java.util.*;
import java.util.stream.*;

List<String> names = List.of("Charlie", "Alice", "Bob", "Dave");

// 自然順（アルファベット昇順）
names.stream()
    .sorted()
    .forEach(System.out::println);
// Alice, Bob, Charlie, Dave

// Comparator で文字数昇順
names.stream()
    .sorted(Comparator.comparingInt(String::length))
    .forEach(System.out::println);
// Bob, Dave, Alice, Charlie

// 逆順
names.stream()
    .sorted(Comparator.reverseOrder())
    .forEach(System.out::println);
// Dave, Charlie, Bob, Alice

// 複数キー（文字数昇順 → アルファベット昇順）
names.stream()
    .sorted(Comparator.comparingInt(String::length)
            .thenComparing(Comparator.naturalOrder()))
    .forEach(System.out::println);

Optional：null安全な値のコンテナ

Stream の findFirst()・min()・max()・reduce()（引数1つ版）は
Optional<T> を返します。値がない場合に null を避けるための仕組みです。

▶ Optional は「値があるかもしれないコンテナ」
✅ 値あり（Optional.of）
Optional<String> opt =
    Optional.of("Hello");
opt.isPresent();  // true
opt.get();        // "Hello"
opt.orElse("?");  // "Hello"
❌ 値なし（Optional.empty）
Optional<String> opt =
    Optional.empty();
opt.isPresent();    // false
opt.get();          // 例外！
opt.orElse("?");    // "?"
メソッド説明
isPresent()値があれば true
isEmpty()値がなければ true（Java 11以降）
get()値を取得（ない場合は例外）
orElse(T)値がない場合のデフォルト値を返す
orElseGet(Supplier)値がない場合に Supplier を実行して返す
orElseThrow()値がない場合に例外を投げる
ifPresent(Consumer)値がある場合だけ Consumer を実行
map(Function)値があれば変換して Optional に包んで返す
Optional の使い方
List<Integer> nums = List.of(3, 1, 4, 1, 5, 9, 2, 6);

// max → Optional<Integer>
Optional<Integer> max = nums.stream().max(Comparator.naturalOrder());

max.ifPresent(v -> System.out.println("最大値: " + v));  // 最大値: 9
System.out.println(max.orElse(-1));                      // 9

// 空のストリームに対する安全な処理
Optional<Integer> empty = Stream.<Integer>empty()
    .max(Comparator.naturalOrder());

System.out.println(empty.isPresent());    // false
System.out.println(empty.orElse(-1));     // -1

// map で値を変換
Optional<String> str = max.map(v -> "最大は " + v + " です");
str.ifPresent(System.out::println);  // 最大は 9 です

flatMap()：ネストを1段階平坦化する

▶ map はリストのリストを返す / flatMap は1つのストリームに平坦化する
["a","b"]
["c","d"]
→ map → [[a,b],[c,d]] ← ネストする
["a","b"]
["c","d"]
→ flatMap →
a, b, c, d ← 平坦化！
flatMap の使い方
// 文章を単語に分割してStream化
List<String> sentences = List.of(
    "Hello World",
    "Stream API is powerful",
    "Java 8"
);

// map だと Stream<String[]> になってしまう
// flatMap で平坦化して Stream<String> にする
List<String> words = sentences.stream()
    .flatMap(s -> Arrays.stream(s.split(" ")))
    .collect(Collectors.toList());

System.out.println(words);
// [Hello, World, Stream, API, is, powerful, Java, 8]

// 重複排除・ソート
words.stream()
    .distinct()
    .sorted()
    .forEach(System.out::println);

コード例：sorted + Optional + flatMap の総合利用

SortOptFlatExample.java
import java.util.*;
import java.util.stream.*;

public class SortOptFlatExample {
    public static void main(String[] args) {
        List<List<Integer>> matrix = List.of(
            List.of(3, 1, 4),
            List.of(1, 5, 9),
            List.of(2, 6, 5)
        );

        // flatMap で1次元に
        List<Integer> flat = matrix.stream()
            .flatMap(Collection::stream)
            .collect(Collectors.toList());
        System.out.println("平坦化: " + flat);

        // ソート + distinct
        List<Integer> sorted = flat.stream()
            .distinct()
            .sorted()
            .collect(Collectors.toList());
        System.out.println("重複除去昇順: " + sorted);

        // Optional で最大・最小
        flat.stream().max(Comparator.naturalOrder())
            .ifPresent(v -> System.out.println("最大: " + v));
        flat.stream().min(Comparator.naturalOrder())
            .ifPresent(v -> System.out.println("最小: " + v));

        // findFirst（条件付き）
        Optional<Integer> firstBig = flat.stream()
            .filter(n -> n > 5)
            .sorted()
            .findFirst();
        System.out.println("5より大きい最小値: " + firstBig.orElse(-1));
    }
}

平坦化: [3, 1, 4, 1, 5, 9, 2, 6, 5]
重複除去昇順: [1, 2, 3, 4, 5, 6, 9]
最大: 9
最小: 1
5より大きい最小値: 6

import java.util.*;
import java.util.stream.*;

public class Q1 {
    public static void main(String[] args) {
        List<String> words = List.of(
            "banana", "Apple", "cherry", "avocado", "Blueberry", "date"
        );

        // 大文字小文字を区別せず昇順
        words.stream()
            .sorted(String.CASE_INSENSITIVE_ORDER)
            .forEach(System.out::println);

        System.out.println("---");

        // 文字数降順 → アルファベット昇順
        words.stream()
            .sorted(Comparator.comparingInt(String::length).reversed()
                    .thenComparing(String.CASE_INSENSITIVE_ORDER))
            .limit(3)
            .forEach(System.out::println);

        System.out.println("---");

        // Optional の操作
        Optional<String> first = words.stream()
            .filter(w -> w.length() == 4)
            .sorted()
            .findFirst();

        System.out.println(first.isPresent());
        System.out.println(first.orElse("なし"));
        System.out.println(first.map(String::toUpperCase).orElse("なし"));
    }
}

解説
  • CASE_INSENSITIVE_ORDER で大文字小文字を無視してアルファベット順にソート。
  • 文字数降順：Blueberry(9) > avocado(7) > banana(6) = cherry(6) > Apple(5) > date(4)。同じ文字数の banana/cherry はアルファベット昇順で banana が先。先頭3件は Blueberry・avocado・banana。
  • 4文字の単語：date のみ。findFirst() で Optional.of("date")。map(String::toUpperCase) で "DATE"。

import java.util.*;
import java.util.stream.*;

public class Q2 {
    public static void main(String[] args) {
        List<List<String>> groups = List.of(
            List.of("Alice", "Bob"),
            List.of("Carol"),
            List.of("Dave", "Eve", "Frank")
        );

        // flatMap で全員を1つのリストに
        List<String> all = groups.stream()
            .flatMap(Collection::stream)
            .collect(Collectors.toList());
        System.out.println("全員: " + all);
        System.out.println("人数: " + all.size());

        // 名前が5文字以上をソートして出力
        all.stream()
            .filter(n -> n.length() >= 5)
            .sorted()
            .forEach(System.out::println);

        System.out.println("---");

        // 各グループのメンバー数
        groups.stream()
            .map(g -> g.size())
            .forEach(System.out::println);

        // 文字を全て取り出す（flatMap）
        long distinctLetters = groups.stream()
            .flatMap(Collection::stream)
            .flatMap(name -> name.chars()
                .mapToObj(c -> (char) c)
                .map(String::valueOf))
            .map(String::toLowerCase)
            .filter(c -> !c.equals(" "))
            .distinct()
            .count();
        System.out.println("ユニーク文字数: " + distinctLetters);
    }
}

問題 11-3-3
コード記述 ⭐⭐⭐

次の条件を満たすクラスを作成してください。

  • Department クラス：name（部署名）と employees（List<String> 社員名リスト）フィールド
  • 複数の部署リストから以下の処理を行う
部署名社員
開発部Alice, Bob, Carol
営業部Dave, Eve
人事部Frank, Grace, Hank
▶ 期待される出力
全社員数: 8
全社員（ソート済み）: [Alice, Bob, Carol, Dave, Eve, Frank, Grace, Hank]
社員数が最多の部署: 開発部 または 人事部
名前が5文字以上の社員: [Alice, Carol, Frank, Grace]
各部署の社員数:
人事部: 3名
営業部: 2名
開発部: 3名
💡 ヒント：全社員は flatMap(d -> d.getEmployees().stream()) で取得します。
社員数が最多の部署は max(Comparator.comparingInt(...)) で求めます。
▶ 答えを見る
Main.java
import java.util.*;
import java.util.stream.*;

public class Main {

    static class Department {
        String       name;
        List<String> employees;
        Department(String name, List<String> employees) {
            this.name=name; this.employees=employees;
        }
        String       getName()      { return name; }
        List<String> getEmployees() { return employees; }
    }

    public static void main(String[] args) {
        List<Department> depts = List.of(
            new Department("開発部", List.of("Alice","Bob","Carol")),
            new Department("営業部", List.of("Dave","Eve")),
            new Department("人事部", List.of("Frank","Grace","Hank"))
        );

        // 全社員数
        long total = depts.stream()
            .flatMap(d -> d.getEmployees().stream())
            .count();
        System.out.println("全社員数: " + total);

        // 全社員（ソート済み）
        List<String> all = depts.stream()
            .flatMap(d -> d.getEmployees().stream())
            .sorted()
            .collect(Collectors.toList());
        System.out.println("全社員（ソート済み）: " + all);

        // 社員数が最多の部署
        depts.stream()
            .max(Comparator.comparingInt(d -> d.getEmployees().size()))
            .ifPresent(d -> System.out.println(
                "社員数が最多の部署: " + d.getName()));

        // 5文字以上の社員
        List<String> longNames = depts.stream()
            .flatMap(d -> d.getEmployees().stream())
            .filter(n -> n.length() >= 5)
            .sorted()
            .collect(Collectors.toList());
        System.out.println("名前が5文字以上の社員: " + longNames);

        // 各部署の社員数
        System.out.println("各部署の社員数:");
        depts.stream()
            .sorted(Comparator.comparing(Department::getName))
            .forEach(d -> System.out.println(
                d.getName() + ": " + d.getEmployees().size() + "名"));
    }
}
解説
  • flatMap で各部署の社員リストを1つの Stream に平坦化します。
  • max(Comparator.comparingInt(d -> d.getEmployees().size())) で社員数最大の部署を取得。開発部と人事部がどちらも3名ですが、Stream は先に来た要素を維持するため開発部が返ります。
  • 5文字以上：Alice(5)・Carol(5)・Frank(5)・Grace(5)（Bob/Dave/Eve/Hank は4文字以下）。
